9 /**
10  * @brief Infinite Impulse Response filter implementation that normalizes the
11  * coefficients upon initialization.
12  *
13  * This class is faster, because each filter iteration involves only addition
14  * and multiplication, no divisions.
15  * It works great for floating point numbers, but might be less ideal
16  * for integer types, because it can create large rounding errors on the
17  * coefficients.
18  *
19  * Implements the following difference equation:
20  *
21  * @f[
22  * y[n] = \frac{1}{a_0} \left(\sum_{i=0}^{N_b-1} b_i \cdot x[n-i]
23  * - \sum_{i=1}^{N_a-1} a_i \cdot y[n-i] \right)
24  * @f]
25  */
26 template <uint8_t NB, uint8_t NA, class T>
27 class NormalizingIIRFilter {
28  public:
29  /**
30  * @brief Construct a new Normalizing IIR Filter object.
31  *
32  * The coefficients @f$ b @f$ and @f$ a @f$ can be derived from the transfer
33  * function:
34  *
35  * @f[
36  * H(z) = \frac{b_0 + b_1 z^{-1} + \ldots + b_{N_b} z ^{-N_b}}
37  * {a_0 + a_1 z^{-1} + \ldots + a_{N_b} z ^{-N_a}}
38  * @f]
39  *
40  * @param b_coefficients
41  * The coefficients of the transfer function numerator.
42  * @param a_coefficients
43  * The coefficients of the transfer function denominator.
44  */
45  NormalizingIIRFilter(const AH::Array<T, NB> &b_coefficients,
46  const AH::Array<T, NA> &a_coefficients)
47  : a0(a_coefficients[0]) {
48  for (uint8_t i = 0; i < 2 * NB - 1; ++i)
49  this->b_coefficients[i] = b_coefficients[(2 * NB - 1 - i) % NB];
50  for (uint8_t i = 0; i < 2 * MA - 1; ++i)
51  this->a_coefficients[i] = a_coefficients[(2 * MA - 2 - i) % MA + 1];
52  }
53 
54  /**
55  * @brief Update the internal state with the new input @f$ x[n] @f$ and
56  * return the new output @f$ y[n] @f$.
57  *
58  * @param input
59  * The new input @f$ x[n] @f$.
60  * @return The new output @f$ y[n] @f$.
61  */
62  T operator()(T input) {
63  // Save the new input to the ring buffer.
64  x[index_b] = input;
65 
66  // Calculate the offset to the shifted coefficients.
67  T *b_coeff_shift = b_coefficients.end() - NB - index_b;
68  T *a_coeff_shift = a_coefficients.end() - MA - index_a;
69 
70  // Multiply and accumulate the inputs and their respective coefficients.
71  T acc = {};
72  for (uint8_t i = 0; i < NB; i++)
73  acc += x[i] * b_coeff_shift[i];
74 
75  // Multiply and accumulate the inputs and their respective coefficients.
76  for (uint8_t i = 0; i < MA; i++)
77  acc -= y[i] * a_coeff_shift[i];
78 
79  // Save the current output
80  acc /= a0;
81  y[index_a] = acc;
82 
83  // Increment and wrap around the index of the ring buffer.
84  index_b++;
85  if (index_b == NB)
86  index_b = 0;
87 
88  index_a++;
89  if (index_a == MA)
90  index_a = 0;
91 
92  return acc;
93  }
94 
95  private:
96  constexpr static uint8_t MA = NA - 1;
97  uint8_t index_b = 0, index_a = 0;
98  AH::Array<T, NB> x = {};
99  AH::Array<T, MA> y = {};
100  AH::Array<T, 2 * NB - 1> b_coefficients;
101  AH::Array<T, 2 * MA - 1> a_coefficients;
102  T a0;
103 };
